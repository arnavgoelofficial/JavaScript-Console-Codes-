var size = 8;
var board = "";

// Outerloop for rows
for (let a = 1; a <= size; a++) {
  // inner loop for columns
  for (let b = 1; b <= size; b++) {
    if ((a+b) % 2 == 0) { // check if col is even
      board += " ";
    } else { // if col is odd
      board += "#";
    }
  }
  board += "\n"; // jump to next row
}
console.log(board);
